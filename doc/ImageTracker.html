def grab_links
  image_urls = URI.open("https://drop.com/buy/drop-olkb-preonic-acute-keycaps") do |f|
    f.read.scan(/.*?<link.*?as="image" href=(".*?").*?/m)
  end
  image_urls.map { |f| /.*(https.*\.png).*/.match(f[0])[1] }
end

def make_links()
  link_tos = grab_links.uniq[0,3].map { |url| "<img src=#{url} />" }

  link_tos.join("\n")
end
